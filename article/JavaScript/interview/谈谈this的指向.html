# 正常情况下,执行函数的方式,就决定了this的指向

  1. 直接调用

function fn (name) {
    console.log(this)
    console.log(this.name)
  }
  fn() // this指向window 输出遥近

# 特殊情况下,this的指向

  1. 箭头函数
function fn1 () {
      console.log(this.name)
      return () => {
        console.log(this.name)
      }
    }

    var name = 'yaojin'

    const obj = {
      name: '遥近'
    }

    const newFn = fn1.call(obj) // 将fn1的this指向obj
    newFn() // 返回的函数是一个箭头函数,它的this指向也是obj
